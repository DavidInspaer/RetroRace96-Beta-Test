# узнать состояние локальных файлов
git status

# чтобы добавить изменения в файлах для коммита
git add _название_файла_или_папки_

# чтобы добавить изменения во всех файлах
git add --all

# сохранить локально все изменения, которые помечены через git add
git commit -m "твой коммент о новой версии"

# так можно назвать разные версии
git tag v1.1

# узнать какие версии вообще есть
git tags

# отправить изменения на сервер
git push

# чтобы скачать свой проект в другую папку
git clone _ссылка_на_реп_

# обновить код с сервака
git pull

# отменить последние действия:
git reset --hard HEAD

# очистка от мусора и сжатие репозитория
git gc --aggressive --prune

# посмотреть изменения без visual studio
gitk
